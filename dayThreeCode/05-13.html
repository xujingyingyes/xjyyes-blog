3、编写parse函数，实现访问对象里属性的值

    let obj = { 
        a: 1, 
        b: { c: 2 }, 
        d: [1, 2, 3], 
        e: [{ f: [4, 5, 6] }] 
    };
    let r1 = parse(obj, 'a');// = 1;
    let r2 = parse(obj, 'b.c');// = 2;
    let r3 = parse(obj, 'd[2]');// = 3;
    let r4 = parse(obj, 'e[0].f[0]');// = 4;

    // 思路
    function parse(obj,str){
        let rs = new Function('obj',`return obj.`+str)
        return rs(obj)
    }
